Test your advanced knowledge of C++ programming language.

1. What is the output of the following code?


                    class Base {

                      public:

                        virtual void show() { cout << "Base"; }

                    };

                    

                    class Derived : public Base {

                      public:

                        void show() { cout << "Derived"; }

                    };

                    

                    int main() {

                      Base* b;

                      Derived d;

                      b = &d;

                      b->show();

                      return 0;

                    }
A. Base
B. Derived
C. BaseDerived
D. Compilation error
Incorrect. The correct output is 'Derived' because 'show' is a virtual function.
Correct! The correct output is 'Derived' because 'show' is a virtual function.
